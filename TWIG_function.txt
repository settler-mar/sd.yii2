_if($is,$then,$else) - аналогичен конструкции $is?$then:$else.  $then и $else не обязательные параметры
_nf($s,$k) -  выполняет ф-ю number_format. $s - входное число. $k - количество знаков после запятой(по умолчанию 2)
_hyphen_email($s) - раставляет ммягкий перенос после @ и )
_hyphen_words($s,$wbr=true) - раставляет мягкие переносы в тексте. $s - строка. $wbr - скрыть тере при разрыве слова (по умолчанию true-скрыть)

_shop_cashback($cashback, $currency = '',$action=0)-функция - вывести кэшбек шопа в списках если нулевой, то сердечки.$action - №акции
_cashback($cashback, $currency = '',$action=0,$mode=0)  - вывести кэшбэк  и валюту, если не задан процента кэшбэка для шопа.$action - №акции. $mode - режим отображние валюты (0 - иконка,1-переменная $currency)
_constant($name) - функция отдать константу по имени
_no_br($content) - функция убрать <br> из контента
_or(...) - вывод первого непустого аргумента
_date - форматирование даты-времени, месяц по-русски

get_menu_item($item) - вывод одного элемента меню врутри <li> ... </li> + дбавляет active если адрес совпадает с ссылкой элемента.
$item=[
   class - стартовый стиль
   href - ссылка
   title - содержание (то что будет между li)
 ]
part($part) - путь для элемента шаблона. к примеру {% include parts('breadcrumbs') %}
